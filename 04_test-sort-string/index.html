JavaScript - Выбор строк по условию, методы - toLowerCase(), toUpperCase(), replace() and slice()

HTML open ->

	<h4>Выбор строк по условию</h4>
	    <i>(Некоторые слова начинаются с букв верхнего регистра)</i>
	    <ol>
	        <li>Happy Birthday!</li>
	        <li>Merry Christmas my love</li>
	        <li>A happy Christmas to all the family</li>
	        <li>You\'re all I want for Christmas</li>
	        <li>Get well soon</li>
	    </ol>
	    <h4>Введите букву, слово или несколько слов из строк выше</h4>
	    <input type="text" class="str">
	    <button>Вывести строку</button>
	    <h4>Вывод строк, в которых есть введенный текст</h4>
	    <i>(Первые буквы строки в верхнем регистре, остальные в нижнем)</i>
	    <div class='output'>
	        <ul></ul>
	    </div>
	    <h4>Метод slice() - возврат нового массива</h4>
	    <p>MAN675847583748sjt567654;Manchester Piccadilly</p>

JavaScript open ->

		var stanzya = document.querySelector('p');
		var btn = document.querySelector('button');
		var input = document.querySelector('input');
		var li = document.querySelectorAll('li');
		
		var list = document.querySelector('.output ul');
		
		function handler() {
			for(var i = 0; i < li.length; i++) {
				var startList = li[i].textContent;
				var inLower = startList.toLowerCase();			// все буквы в нижний регистр
				var inUpper = inLower[0].toUpperCase();			// первая буква в строке в верхний регистр
				var inReplace = inLower.replace(inLower[0], inUpper[0]);	// Меняем первый символ в верхний регистр		
				if(li[i].textContent.indexOf(input.value) !== -1) {		// Если индекс не -1 (не пусто)
					if(input.value !== '') {
						var listItem = document.createElement('li');
						listItem.textContent = inReplace;
						list.appendChild(listItem);
					}
				}
			}
		}
		
		btn.addEventListener('click', handler);
		
		// Создаем строку
		var stanzya = 'MAN675847583748sjt567654;Manchester Piccadilly';
		// Выбираем три символа с начала строки
		var code = stanzya.slice(0,3);
		// Выбираем символы с 25-го до последнего
		var codeSlice = stanzya.slice(25);
		// Создаем новую строку
		var result = code+': '+codeSlice;
		// Выводим результат
		document.write(result);
