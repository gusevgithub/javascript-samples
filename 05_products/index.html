JavaScript - Работа с массивом данных, использование методов split() and toFixed()

HTML open ->

	 <div class='output'>
	        <ul></ul>
	 </div>

	 <div class='output'>
	        <p></p>
	 </div>

JavaScript open ->

	var list = document.querySelector('.output ul');
	var totalBox = document.querySelector('.output p');
	var total = 0;
	list.innerHTML = '';
	totalBox.textContent = '';
	var products = ['Underpants:6.99',
			'Socks:5.99',
			'T-shirt:14.99',
			'Trousers:31.99',
			'Shoes:23.99'];

	for (var i = 0; i < products.length; i++) {
	/* // Вариант - 1
		var myIndexOf = products[i].indexOf(':');
		var name = products[i].slice(0,myIndexOf);
		var price = Number(products[i].slice(myIndexOf+1));
	*/
	// Вариант -2
	var subArray = products[i].split(':');	// Разбиение свойства и значение на массив
	var name = subArray[0];			// Первое значение массива - name
	var price = Number(subArray[1]);	// Второе значение массива - price
	total += price;				// Добавляем вторые значения массива в переменную total
	itemText = name+' - $'+price;		// Создаем новые строки - itemText

	var listItem = document.createElement('li');	// Создаем новые элементы - li
	listItem.textContent = itemText;		// Добавляем в элементы полученные строки - itemText
	list.appendChild(listItem);			// Элементы li будут дочерними элементами .output ul
	}

	totalBox.textContent = 'Total: $' + total.toFixed(2);	// Метод toFixed() форматирует число, 2 - сотые доли
